HTML Starting Point

Basic HTML starting template that defines meta data, CSS file, Javascript file, and favicon image.

<!DOCTYPE html>
  <html>
    <head>
        <meta charset="utf-8">
        <title>Zen CSS | CSS Framework </title>
        <meta name="author" content="Kyle Fleishman">
        <meta name="description" content="CSS Styling Framework">
        <meta name="viewport" content="width=device-width, intitial-scale=1">
        <link href="https://fonts.googleapis.com/css2?family=Carme&display=swap" rel="stylesheet">
        <link rel="stylesheet" href="css/modern-normalize.css">
        <link rel="stylesheet" href="css/clarity.css">
        <link rel="icon" href="assets/image1.png">
      </head>
         <body>
        </body>
  </html>
